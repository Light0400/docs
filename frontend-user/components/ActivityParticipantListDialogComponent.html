<ng-template #itemRight let-data>
    @if(actUser?.IsOrganizer || actUser?.IsLeadRider){
    @if(activity?.LeaderStartTime){
    <!-- #region 已有領騎者開始騎乘, 顯示騎乘狀態 -->
    @if(data.item.Record?.Stop){
    <div class="row tmup-align-center tmup-gap-4 tmup-fc-gray-3 tmup-fs-14 t-bold-500">
        <mat-icon svgIcon="pause-circle"></mat-icon>
        <div>
            {{ 'pages.activity-join-list.stop' | translate}}
        </div>
    </div>
    }@else if(!!data.item.StartTime && !data.item.CompletedTime){
    <div class="row tmup-align-center tmup-gap-4 tmup-fc-teal tmup-fs-14 t-bold-500">
        <mat-icon svgIcon="start"></mat-icon>
        <div>
            {{ 'pages.activity-join-list.running' | translate}}
        </div>
    </div>
    }@else if(!!data.item.CompletedTime){
    <div class="row tmup-align-center tmup-gap-4 tmup-fc-blue tmup-fs-14 t-bold-500">
        <div>
            {{ 'pages.activity-join-list.ended' | translate}}
        </div>
    </div>
    }@else{
    <div class="row tmup-align-center tmup-gap-4 tmup-fc-gray-5 tmup-fs-14 t-bold-500">
        <div>
            {{ 'pages.activity-join-list.not-start' | translate}}
        </div>
    </div>
    }
    <!-- #endregion -->

    }@else{
    <!-- #region 尚未開始騎乘, 顯示是否就緒 -->

    @if(data.isGuest){

    <!-- #region 訪客 -->
    @if(data.item.IsReady){

    <button type="button" class="custom active row tmup-center" (click)="setReady(data.item, !data.item.IsReady)">
        <mat-icon>check</mat-icon>
    </button>

    }@else{

    <button type="button" class="custom" (click)="setReady(data.item, !data.item.IsReady)">
        {{ 'common.checkin' | translate }}
    </button>

    }
    <!-- #endregion -->

    }@else{

    <!-- #region 建立者或領騎者 -->
    @if(data.item.IsReady){

    <div>
        <button type="button" class="custom active row tmup-center" (click)="setReady(data.item, !data.item.IsReady)">
            <mat-icon>check</mat-icon>
            <span>{{'common.readied'|translate}}</span>
        </button>
    </div>

    }@else{

    <div>
        <button type="button" class="custom" (click)="setReady(data.item, !data.item.IsReady)">
            {{'common.ready' | translate}}
        </button>
    </div>

    }
    <!-- #endregion -->

    }
    <!-- #endregion -->

    }
    }
</ng-template>
