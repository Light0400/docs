<label [htmlFor]="'label-search-input-' + id">
  @if(prefixIcon){
  <mat-icon class="search-icon" [svgIcon]="prefixIcon"></mat-icon>
  }

  <input [id]="'label-search-input-' + id" #searchInput type="text" [placeholder]="placeholder ?? ''"
    [formControl]="searchCtrl" (focus)="_onFocus($event)" (blur)="_onBlur($event)" />
    
  @if(maxWords != null && maxWords > 0){
  <div class="max-words">
    {{ searchCtrl.value?.trim()?.length }}/{{ maxWords }}
  </div>
  }

  @if(showClearBtn){
  <div [hidden]="!searchCtrl.value?.trim()?.length" (click)="onClearSearch()">
    <mat-icon svgIcon="clear" class="fs-16"></mat-icon>
  </div>
  }
</label>
