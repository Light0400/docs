  <lib-header [show]="true" [isDialog]="true" [backNotShow]="true">
    <ng-container ngProjectAs="[h-title]">
      {{ "dialog.stopwatch-info.title" | translate }}
    </ng-container>

    <ng-container ngProjectAs="[h-right]">
      <button class="tmup-btn-icon" type="button" (click)="back()">
        <mat-icon svgIcon="close"></mat-icon>
      </button>
    </ng-container>
  </lib-header>

  <main role="main">
    @for(item of options; track item.id){
    <div class="item c-pointer" (click)="onPressItem(item)">
      <div class="info column">
        <span class="title">
          {{ item.text | translate }}
        </span>

        @if(item.type === 'extra'){
        <span class="subtitle">
          @if(!deviceBindings[item.id]) {
          <ng-container>
            {{ "dialog.stopwatch-info.need-link-device" | translate : ({ deviceName: (item.text | translate) }) }}
          </ng-container>
          } @else {
          <ng-container>
            {{ "dialog.stopwatch-info.linked-device" | translate : ({ deviceName: deviceBindings[item.id]?.name}) }}
          </ng-container>
          }
        </span>
        }
      </div>
      @if(item.id === checked){
      <mat-icon class="checked" svgIcon="fill-check"></mat-icon>
      } @else {
      <mat-icon svgIcon="checkbox"></mat-icon>
      }
    </div>
    }

  </main>
</lib-draggable-list>
